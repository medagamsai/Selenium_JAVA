public class ToUseMultiselect_Deselect {
	public static void main(String[] args) throws InterruptedException {
		WebDriver driver = new ChromeDriver();
		driver.manage().window().maximize();
		driver.manage().timeouts().implicitlyWait(Duration.ofSeconds(15));
		
		driver.get("https://demoapp.skillrary.com/");
		
		//identify listbox
		WebElement listbox = driver.findElement(By.id("cars"));
		
		//to handle listbox
		Select multihandle = new Select(listbox);
		System.out.println(multihandle.isMultiple());
		
		//call methods
		multihandle.selectByIndex(0);
		multihandle.selectByValue("99");
		Thread.sleep(2000);
		
		// deselect it
//		multihandle.deselectByIndex(0);
//		multihandle.deselectByValue("99");
		multihandle.deselectAll();
		
		driver.quit();
	}
